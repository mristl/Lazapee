# Binary search function def binary_search(arr, target, key=lambda x: x): left, right = 0, len(arr) - 1 while left <= right: mid = (left + right) // 2 mid_value = key(arr[mid]) if mid_value == target: return arr[mid] elif mid_value < target: left = mid + 1 else: right = mid - 1 return None # Return None if the item isn't found # Sample list of books books = [ Book(title="Harry Potter", author="J.K. Rowling"), Book(title="Lord of the Rings", author="J.R.R. Tolkien"), Book(title="The Hobbit", author="J.R.R. Tolkien"), ] # Sort the list of books by title sorted_books = sorted(books, key=lambda book: book.title) # Test search for a specific title search_title = "The Hobbit" result = binary_search(sorted_books, search_title, key=lambda book: book.title) if result: print(f"Found: {result.title}") else: print("Not found.")
